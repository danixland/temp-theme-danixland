Feel free to adjust your CSS/SCSS as desired. */}} {{- $imgClass := "w-full h-auto animate-fade" -}} {{- $dataSzes := "(min-width: 1024px) 100vw, 50vw" -}} {{/* Now we'll create the 20-pixel-wide LQIP and turn it into Base64-encoded data, which is better for performance and caching. */}} {{- $LQIP_img := $source.Resize "20x jpg" -}} {{- $LQIP_b64 := $LQIP_img.Content | base64Encode -}} {{/* $CFPstyle is for use in styling the div's background, as you'll see shortly. */}} {{- $CFPstyle := printf "%s%s%s%v%s%v%s" "background: url(data:image/jpeg;base64," $LQIP_b64 "); background-size: cover; background-repeat: no-repeat; width: " $source.Width "px; height: " $source.Height "px;" -}} {{/* Then, we create a 640-pixel-wide JPG of the image.
